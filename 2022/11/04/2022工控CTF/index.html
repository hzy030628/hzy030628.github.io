   return a  
def getpq(n,e,d):
    p = 1  
    q = 1  
    for i in range(1,101):
        k = d * e - 1  
        g = random.randint ( 0 , n )  
        while p==1 and q==1 and k % 2 == 0:

            y = pow(g,k,n)

                q = n//p
    return p,q
