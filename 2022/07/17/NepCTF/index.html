import zipfile
import re

zipname = "C:\\Users\\25963\\Desktop\\新\\xxx\\232\\231\\"+"230.zip"

usb.transfer_type == 0x01 and usb.src == "1.5.1"

tshark -r 2.pcap -T fields -e usb.capdata > usbdata.txt

f=open('usbdata.txt','r')

                    out+=a[i]+a[i+1]+":"
                else:
                    out+=a[i]+a[i+1]
            fi.write(out)
            fi.write('\n')
    else:
        break

fi.close()
