with open('C:/Users/25963/Desktop/00000190/Dic/Reverse.piz', 'rb') as f:

  s = f.read()

 

t = [x // 16 + 16 * (x % 16) for x in s]

with open('C:/Users/25963/Desktop/00000190/Dic/x.zip', 'wb') as f:
